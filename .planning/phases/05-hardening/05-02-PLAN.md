export type RedisLike = {
  set(key: string, value: unknown, opts?: { nx?: true; ex?: number }): Promise<unknown>;
  get(key: string): Promise<unknown>;
  getdel(key: string): Promise<unknown>;
  del(...keys: string[]): Promise<unknown>;
};
export async function markEventOnce(redis, eventId, ttl): Promise<boolean> {
  const result = await redis.set(`evt:${eventId}`, 1, { nx: true, ex: ttl });
  return result !== null; // false on a Slack retry — already drops redeliveries
}
export async function markWebhookDeliveryOnce(redis, key, ttl): Promise<boolean> { /* "whk:" NX EX */ }
